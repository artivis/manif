110 template <typename _Derived>
111 typename SE2Base<_Derived>::Tangent
112 SE2Base<_Derived>::lift(OptJacobianRef J_t_m) const
113 {
114  using std::abs;
115  using std::cos;
116  using std::sin;
117 
118  const Scalar theta = angle();
119  const Scalar cos_theta = coeffs()[2];
120  const Scalar sin_theta = coeffs()[3];
121  const Scalar theta_sq = theta * theta;
122 
123  Scalar A, // sin_theta_by_theta
124  B; // one_minus_cos_theta_by_theta
125 
126  if (abs(theta) < Constants<Scalar>::eps)
127  {
128  // Taylor approximation
129  A = Scalar(1) - Scalar(1. / 6.) * theta_sq;
130  B = Scalar(.5) * theta - Scalar(1. / 24.) * theta * theta_sq;
131  }
132  else
133  {
134  // Euler
135  A = sin_theta / theta;
136  B = (Scalar(1) - cos_theta) / theta;
137  }
138 
139  const Scalar den = Scalar(1) / (A*A + B*B);
140 
141  A *= den;
142  B *= den;
143 
144  Tangent tan( A * x() + B * y(),
145  -B * x() + A * y(),
146  theta );
147 
148  if (J_t_m)
149  {
150  // Jr^-1
151  (*J_t_m) = tan.rjac().inverse();
152  }
153 
154  return tan;
155 }

188  lhs_imag * m_se2.x() + lhs_real * m_se2.y() + y(),
189  lhs_real * rhs_real - lhs_imag * rhs_imag,
