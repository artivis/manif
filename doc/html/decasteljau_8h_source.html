37  // Number of connected, non-overlapping segments
38  const unsigned int n_segments =
39  std::floor(double(trajectory.size()-degree)/(degree-1)+1);
40 
41  std::vector<std::vector<const LieGroup*>> segments_control_points;
42  for (unsigned int t=0; t<n_segments; ++t)
43  {
44  segments_control_points.emplace_back(std::vector<const LieGroup*>());
45 
46  // Retrieve control points of the current segment
47  for (int n=0; n<degree; ++n)
48  {
49  segments_control_points.back().push_back( &trajectory[t*(degree-1)+n] );
50  }
51  }
52 
53  // Close the curve if there are left-over points
54  if (closed_curve && (n_segments*(degree-1)) <= trajectory.size()-1)
55  {
56  const int last_pts_idx = n_segments*(degree-1);
57  const int left_over = trajectory.size()-1-last_pts_idx;
58  segments_control_points.emplace_back(std::vector<const LieGroup*>());
59 
60  // Get the left-over points
61  for (int p=last_pts_idx; p<trajectory.size(); ++p)
62  {
63  segments_control_points.back().push_back( &trajectory[p] );
64  }
65  // Add a extra points from the beginning of the trajectory
66  for (int p=0; p<degree-left_over-1; ++p)
67  {
68  segments_control_points.back().push_back( &trajectory[p] );
69  }
70  }
71 
72  const int segment_k_interp = (degree == 2) ?
73  k_interp : k_interp * degree;
74 
75  // Actual curve fitting
76  std::vector<LieGroup> curve;
77  for (unsigned int s=0; s<segments_control_points.size(); ++s)
78  {
79  for (int t=1; t<=segment_k_interp; ++t)
80  {
81  // t in [0,1]
82  const double t_01 = static_cast<double>(t)/(segment_k_interp);
83 
84  std::vector<LieGroup> Qs, Qs_tmp;
85 
86  for (const auto m : segments_control_points[s])
87  Qs.emplace_back(*m);
88 
89  // recursive chunk of the algo,
90  // compute tmp control points.
91  for (int i=0; i<degree-1; ++i)
92  {
93  for (int q=0; q<Qs.size()-1; ++q)
94  {
95  Qs_tmp.push_back( Qs[q].rplus(Qs[q+1].rminus(Qs[q]) * t_01) );
96  }
97 
98  Qs = Qs_tmp;
99  Qs_tmp.clear();
100  }
101 
102  curve.push_back(Qs[0]);
103  }
104  }
