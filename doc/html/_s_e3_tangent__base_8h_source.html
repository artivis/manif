124 template <typename _Derived>
125 typename SE3TangentBase<_Derived>::Jacobian
126 SE3TangentBase<_Derived>::rjac() const
127 {
128  using std::cos;
129  using std::sin;
130  using std::sqrt;
131 
133  Jacobian Jr = Jacobian::Zero();
134  Jr.template topLeftCorner<3,3>() = asSO3().rjac();
135  Jr.template bottomRightCorner<3,3>() =
136  Jr.template topLeftCorner<3,3>();
137 
138  const Scalar theta_sq = asSO3().coeffs().squaredNorm();
139  const LieAlg V = skew(v());
140  const LieAlg W = asSO3().hat();
141 
142  Scalar A(0.5), B, C, D;
143 
144  // Small angle approximation
145  if (theta_sq <= Constants<Scalar>::eps_s)
146  {
147  B = Scalar(1./6.) + Scalar(1./120.) * theta_sq;
148  C = -Scalar(1./24.) + Scalar(1./720.) * theta_sq;
149  D = -Scalar(1./60.);
150  }
151  else
152  {
153  const Scalar theta = sqrt(theta_sq);
154  const Scalar sin_theta = sin(theta);
155  const Scalar cos_theta = cos(theta);
156 
157  B = (theta - sin_theta) / (theta_sq*theta);
158  C = (Scalar(1) - theta_sq/Scalar(2) - cos_theta) / (theta_sq*theta_sq);
159  D = (C - Scalar(3)*(theta-sin_theta-theta_sq*theta/Scalar(6)) / (theta_sq*theta_sq*theta));
160 
161  // http://asrl.utias.utoronto.ca/~tdb/bib/barfoot_ser17_identities.pdf
162 // C = (theta_sq+Scalar(2)*cos_theta-Scalar(2)) / (Scalar(2)*theta_sq*theta_sq);
163 // D = (Scalar(2)*theta - Scalar(3)*sin_theta + theta*cos_theta) / (Scalar(2)*theta_sq*theta_sq*theta);
164  }
165 
168  Jr.template topRightCorner<3,3>().noalias() =
169  - A * V
170  + B * (W*V + V*W - (W*V)*W)
171  + C * ((W*W)*V + (V*W)*W - Scalar(3)*(W*V)*W)
172  - D * Scalar(0.5) * (((W*V)*W)*W + ((W*W)*V)*W);
173 
174  return Jr;
175 }

178 typename SE3TangentBase<_Derived>::Jacobian
179 SE3TangentBase<_Derived>::ljac() const
180 {
181  using std::cos;
182  using std::sin;
183  using std::sqrt;
184 
186  Jacobian Jl = Jacobian::Zero();
187  Jl.template topLeftCorner<3,3>() = asSO3().ljac();
188  Jl.template bottomRightCorner<3,3>() =
189  Jl.template topLeftCorner<3,3>();
190 
191  const Scalar theta_sq = asSO3().coeffs().squaredNorm();
192  const LieAlg V = skew(v());
193  const LieAlg W = asSO3().hat();
194 
195  Scalar A(0.5), B, C, D;
196 
197  // Small angle approximation
198  if (theta_sq <= Constants<Scalar>::eps_s)
199  {
200  B = Scalar(1./6.) + Scalar(1./120.) * theta_sq;
201  C = -Scalar(1./24.) + Scalar(1./720.) * theta_sq;
202  D = -Scalar(1./60.);
203  }
204  else
205  {
206  const Scalar theta = sqrt(theta_sq);
207  const Scalar sin_theta = sin(theta);
208  const Scalar cos_theta = cos(theta);
209 
210  B = (theta - sin_theta) / (theta_sq*theta);
211  C = (Scalar(1) - theta_sq/Scalar(2) - cos_theta) / (theta_sq*theta_sq);
212  D = (C - Scalar(3)*(theta-sin_theta-theta_sq*theta/Scalar(6)) / (theta_sq*theta_sq*theta));
213 
214  // http://asrl.utias.utoronto.ca/~tdb/bib/barfoot_ser17_identities.pdf
215 // C = (theta_sq+Scalar(2)*cos_theta-Scalar(2)) / (Scalar(2)*theta_sq*theta_sq);
216 // D = (Scalar(2)*theta - Scalar(3)*sin_theta + theta*cos_theta) / (Scalar(2)*theta_sq*theta_sq*theta);
217  }
218 
220  Jl.template topRightCorner<3,3>().noalias() =
221  + A * V
222  + B * (W*V + V*W + (W*V)*W)
223  - C * ((W*W)*V + (V*W)*W - ((Scalar(3)*W)*V)*W)
224  - D * Scalar(0.5) * (((W*V)*W)*W + ((W*W)*V)*W);
225
