74 static typename LieGroupBase<_Derived>::LieGroup
75 interpolate_slerp(const LieGroupBase<_Derived>& ma,
76  const LieGroupBase<_Derived>& mb,
77  const _Scalar t,
78  typename LieGroupBase<_Derived>::OptJacobianRef J_mc_ma = LieGroupBase<_Derived>::_,
79  typename LieGroupBase<_Derived>::OptJacobianRef J_mc_mb = LieGroupBase<_Derived>::_)
80 {
81  MANIF_CHECK(t >= _Scalar(0) && t <= _Scalar(1),
82  "s must be be in [0, 1].");
83 
84  using LieGroup = typename LieGroupBase<_Derived>::LieGroup;
85  using Jacobian = typename LieGroupBase<_Derived>::Jacobian;
86 
87  LieGroup mc;
88 
89  const auto _ = LieGroupBase<_Derived>::_;
90 
92  if (J_mc_ma && J_mc_mb)
93  {
94  Jacobian J_rmin_ma, J_rmin_mb;
95  Jacobian p1J_mc_ma;
96  Jacobian J_mc_rmin;
97 
98  mc = ma.rplus( mb.rminus(ma, J_rmin_mb, J_rmin_ma) * t, p1J_mc_ma, J_mc_rmin);
99 
100  (*J_mc_ma) = p1J_mc_ma + J_mc_rmin * (J_rmin_ma * t);
101  (*J_mc_mb) = J_mc_rmin * (J_rmin_mb * t);
102  }
103  else if (J_mc_ma)
104  {
105  Jacobian J_rmin_ma, p1J_mc_ma;
106  Jacobian J_mc_rmin;
107 
108  mc = ma.rplus( mb.rminus(ma, _, J_rmin_ma) * t, p1J_mc_ma, J_mc_rmin);
109 
110  (*J_mc_ma) = p1J_mc_ma + J_mc_rmin * (J_rmin_ma * t);
111  }
112  else if (J_mc_mb)
113  {
114  Jacobian J_rmin_mb, J_mc_rmin;
115 
116  mc = ma.rplus( mb.rminus(ma, J_rmin_mb, _) * t, _, J_mc_rmin);
117 
118  (*J_mc_mb) = J_mc_rmin * (J_rmin_mb * t);
119  }
120  else
121  {
122  mc = ma.rplus( mb.rminus(ma) * t );
123  }
124 
125  return mc;
126 }
