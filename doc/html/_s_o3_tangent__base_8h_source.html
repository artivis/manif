52 template <typename _Derived>
53 typename SO3TangentBase<_Derived>::LieGroup
54 SO3TangentBase<_Derived>::retract(OptJacobianRef J_m_t) const
55 {
56  using std::sqrt;
57  using std::cos;
58  using std::sin;
59 
60  const DataType& theta_vec = coeffs();
61  const Scalar theta_sq = theta_vec.squaredNorm();
62  const Scalar theta = sqrt(theta_sq);
63 
64  if (theta_sq > Constants<Scalar>::eps_s)
65  {
66  if (J_m_t)
67  {
68  Jacobian M1, M2;
69 
70  const LieAlg W = hat();
71 
72  M1.noalias() = (Scalar(1.0) - cos(theta)) / theta_sq * W;
73  M2.noalias() = (theta - sin(theta)) / (theta_sq * theta) * (W * W);;
74 
75  *J_m_t = Jacobian::Identity() - M1 + M2;
76  }
77 
78  return LieGroup( Eigen::AngleAxis<Scalar>(theta, theta_vec.normalized()) );
79  }
80  else
81  {
82  if (J_m_t)
83  {
84  *J_m_t = Jacobian::Identity() - Scalar(0.5) * hat();
85  }
86 
87  return LieGroup(x()/Scalar(2), y()/Scalar(2), z()/Scalar(2), Scalar(1));
88  }

99 typename SO3TangentBase<_Derived>::Jacobian
100 SO3TangentBase<_Derived>::ljac() const
101 {
102  using std::sqrt;
103  using std::cos;
104  using std::sin;
105 
106  const Scalar theta_sq = coeffs().squaredNorm();
107 
108  const LieAlg W = hat();
109 
110  // Small angle approximation
111  if (theta_sq <= Constants<Scalar>::eps_s)
112  return Jacobian::Identity() - Scalar(0.5) * W;
113 
114  const Scalar theta = sqrt(theta_sq); // rotation angle
115  Jacobian M1, M2;
116  M1.noalias() = (Scalar(1) - cos(theta)) / theta_sq * W;
117  M2.noalias() = (theta - sin(theta)) / (theta_sq * theta) * (W * W);
118 
119  return Jacobian::Identity() + M1 + M2;
120 }

131 SO3TangentBase<_Derived>::ljacinv() const

134  using std::cos;
135  using std::sin;

137  const Scalar theta_sq = coeffs().squaredNorm();
138 
139  const LieAlg W = hat();
140 
141  if (theta_sq <= Constants<Scalar>::eps_s)
142  return Jacobian::Identity() + Scalar(0.5) * W;
143 
144  const Scalar theta = sqrt(theta_sq); // rotation angle
145  Jacobian M;
146  M.noalias() = (Scalar(1) / theta_sq -
