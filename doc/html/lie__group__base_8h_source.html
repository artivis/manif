358 template <typename _Derived>
359 template <typename _DerivedOther>
360 typename LieGroupBase<_Derived>::Tangent
361 LieGroupBase<_Derived>::lminus(
362  const LieGroupBase<_DerivedOther>& m,
363  OptJacobianRef J_t_ma,
364  OptJacobianRef J_t_mb) const
365 {
366  Tangent t;
367 
369  if (J_t_ma && J_t_mb)
370  {
371  Jacobian J_inv_mb;
372  Jacobian J_comp_inv;
373  Jacobian J_comp_ma;
374  Jacobian J_t_comp;
375 
376  t = compose(m.inverse(J_inv_mb),
377  J_comp_ma, J_comp_inv).lift(J_t_comp);
378 
379  J_t_ma->noalias() = J_t_comp * J_comp_ma;
380  J_t_mb->noalias() = J_t_comp * J_comp_inv * J_inv_mb;
381  }
382  else if (J_t_ma && !J_t_mb)
383  {
384  Jacobian J_comp_a;
385  Jacobian J_t_comp;
386 
387  t = compose(m.inverse(),
388  J_comp_a, _ ).lift(J_t_comp);
389 
390  J_t_ma->noalias() = J_t_comp * J_comp_a;
391  }
392  else if (!J_t_ma && J_t_mb)
393  {
394  Jacobian J_inv_mb;
395  Jacobian J_comp_inv;
396  Jacobian J_t_comp;
397 
398  t = compose(m.inverse(J_inv_mb),
399  _, J_comp_inv).lift(J_t_comp);
400 
401  J_t_mb->noalias() = J_t_comp * J_comp_inv * J_inv_mb;
402  }
403  else
404  {
405  t = compose(m.inverse()).lift();
406  }
407 
408  return t;
409 }
